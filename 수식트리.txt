#include<stdio.h>
#include<stdlib.h>

typedef struct TreeNode {
    char data;
    struct TreeNode *left, *right;
}TreeNode;

#define SIZE 100
int top=-1;

TreeNode n1={10, NULL, NULL};
TreeNode n2={20, NULL, NULL};
TreeNode n3={'*', &n1, &n2};
TreeNode n4={30, NULL, NULL};
TreeNode n5={40, NULL, NULL};
TreeNode n6={'/', &n4, &n5};
TreeNode n7={'+', &n3, &n6};
TreeNode *root =&n7;

int evaluate(TreeNode *root)
{
    if (root == NULL)
    return 0;
    if (root->left == NULL && root->right == NULL)
    return root->data;
    else{
        int op1 = evaluate(root->left);
        int op2 = evaluate(root->right);
        printf("%d %c %d를 계산\n", op1, root->data, op2);
        switch (root->data) {
            case '+':
            return op1 + op2;
            case '-':
            return op1-op2;
            case '*':
            return op1*op2;
            case '/':
            return op1/op2;
        }
    }
    return 0;
}



int main(void)
{
    printf("수식의 값은 %d입니다", evaluate(root));
    return 0;
}
